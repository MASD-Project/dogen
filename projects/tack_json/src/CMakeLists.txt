set(all_files "")
file(GLOB_RECURSE all_files RELATIVE
    "${CMAKE_CURRENT_SOURCE_DIR}/"
    "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp")

add_library(tack_json STATIC ${all_files})
target_link_libraries(tack_json
    tack
    config
    dynamic
    utility
    ${LIBXML2_LIBRARIES}
    ${Boost_LIBRARIES})

set_target_properties(tack_json PROPERTIES OUTPUT_NAME dogen_tack_json)

install(TARGETS tack_json ARCHIVE DESTINATION lib COMPONENT libraries)
